#include int main() { int i, n, time, remain, tq; int wt = 0, tat = 0; printf("Enter number of processes: "); scanf("%d", &n); remain = n; int bt[n], rt[n]; for(i = 0; i < n; i++) { printf("Enter Burst Time for Process %d: ", i+1); scanf("%d", &bt[i]); rt[i] = bt[i]; } printf("Enter Time Quantum: "); scanf("%d", &tq); time = 0; i = 0; while(remain != 0) { if(rt[i] <= tq && rt[i] > 0) { time += rt[i]; rt[i] = 0; remain--; printf("Process P%d completed\n", i+1); tat += time; wt += time - bt[i]; } else if(rt[i] > 0) { rt[i] -= tq; time += tq; } if(i == n-1) i = 0; else i++; } printf("\nAverage Waiting Time = %.2f", (float)wt/n); printf("\nAverage Turnaround Time = %.2f\n", (float)tat/n); return 0; } #!/bin/bash echo "Enter Principal Amount:" read p echo "Enter Rate of Interest:" read r echo "Enter Time (in years):" read t # Simple Interest si=$((p*r*t/100)) # Compound Interest amount=$(echo "$p*(1+$r/100)^$t" | bc -l) ci=$(echo "$amount-$p" | bc -l) echo "Simple Interest = $si" echo "Compound Interest = $ci"
